defmodule NeuralMeshLite.CasteSystem do
  @moduledoc """
  Emergent caste specialization using the Response Threshold Model.
  Minds self-organize into workers, scouts, and queens based on internal thresholds
  and environmental stimuli.
  """
  use GenServer
  require Logger

  defstruct [
    :thresholds,        # %{mind_id => %{task_type => threshold}}
    :stimuli,           # %{task_type => current_stimulus_level}
    :castes,            # %{mind_id => current_caste}
    :quorum_threshold   # For collective decision-making
  ]

  @caste_types [:worker, :scout, :queen]
  @task_types [:explore, :exploit, :rest, :broadcast, :evolve]

  # Client API
  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def register_mind(mind_id) do
    GenServer.cast(__MODULE__, {:register, mind_id})
  end

  def update_stimulus(task_type, delta) do
    GenServer.cast(__MODULE__, {:update_stimulus, task_type, delta})
  end

  def get_caste(mind_id) do
    GenServer.call(__MODULE__, {:get_caste, mind_id})
  end

  def select_action(mind_id) do
    GenServer.call(__MODULE__, {:select_action, mind_id})
  end

  def trigger_quorum(proposal) do
    GenServer.call(__MODULE__, {:quorum, proposal})
  end

  # Server callbacks
  @impl true
  def init(_opts) do
    state = %__MODULE__{
      thresholds: %{},
      stimuli: Map.new(@task_types, &{&1, 0.5}),
      castes: %{},
      quorum_threshold: 0.6
    }
    # Decay stimuli over time
    schedule_decay()
    {:ok, state}
  end

  @impl true
  def handle_cast({:register, mind_id}, state) do
    # Assign random initial thresholds (genetic predisposition)
    thresholds = Map.new(@task_types, fn task ->
      {task, :rand.uniform()}
    end)
    
    # Determine initial caste based on threshold pattern
    caste = determine_caste(thresholds)
    
    state = %{state | 
      thresholds: Map.put(state.thresholds, mind_id, thresholds),
      castes: Map.put(state.castes, mind_id, caste)
    }
    
    Logger.debug("Mind #{mind_id} registered as #{caste}")
    {:noreply, state}
  end

  def handle_cast({:update_stimulus, task_type, delta}, state) do
    stimuli = Map.update!(state.stimuli, task_type, &max(0.1, min(1.0, &1 + delta)))
    {:noreply, %{state | stimuli: stimuli}}
  end

  @impl true
  def handle_call({:get_caste, mind_id}, _from, state) do
    {:reply, Map.get(state.castes, mind_id), state}
  end

  def handle_call({:select_action, mind_id}, _from, state) do
    thresholds = Map.get(state.thresholds, mind_id, %{})
    
    # Response threshold model: probability of engaging in task increases
    # as stimulus exceeds threshold
    action_probs = Map.new(@task_types, fn task ->
      stimulus = Map.get(state.stimuli, task, 0.5)
      threshold = Map.get(thresholds, task, 0.5)
      prob = if stimulus > threshold do
        (stimulus - threshold) / (1 - threshold)
      else
        0.01  # Small baseline probability
      end
      {task, prob}
    end)
    
    # Normalize and sample
    total = Enum.sum(Map.values(action_probs))
    if total > 0 do
      r = :rand.uniform() * total
      {action, _} = Enum.reduce_while(action_probs, {nil, r}, fn {task, prob}, {_, remaining} ->
        if remaining <= prob do
          {:halt, {task, remaining}}
        else
          {:cont, {nil, remaining - prob}}
        end
      end)
      {:reply, action || :rest, state}
    else
      {:reply, :rest, state}
    end
  end

  def handle_call({:quorum, proposal}, _from, state) do
    # Quorum sensing: track how many minds support this proposal
    # (Simplified - in practice, would aggregate votes over time window)
    supporters = Enum.count(state.castes, fn {_, caste} -> 
      caste_supports?(caste, proposal)
    end)
    
    total = map_size(state.castes)
    quorum_reached = if total > 0 do
      supporters / total >= state.quorum_threshold
    else
      false
    end
    
    {:reply, %{reached: quorum_reached, supporters: supporters, total: total}, state}
  end

  @impl true
  def handle_info(:decay, state) do
    # Gradually reduce stimuli (pheromone evaporation)
    stimuli = Map.new(state.stimuli, fn {task, level} ->
      {task, level * 0.99}
    end)
    schedule_decay()
    {:noreply, %{state | stimuli: stimuli}}
  end

  # Private helpers
  defp determine_caste(thresholds) do
    explore_threshold = Map.get(thresholds, :explore, 0.5)
    broadcast_threshold = Map.get(thresholds, :broadcast, 0.5)
    
    cond do
      explore_threshold < 0.3 -> :scout
      broadcast_threshold < 0.3 -> :queen
      true -> :worker
    end
  end

  defp caste_supports?(:queen, :evolve), do: true
  defp caste_supports?(:scout, :explore), do: true
  defp caste_supports?(:worker, :exploit), do: true
  defp caste_supports?(_, _), do: false

  defp schedule_decay do
    Process.send_after(self(), :decay, 10_000)
  end
end

defmodule NeuralMeshLite.PheromoneTrails do
  @moduledoc """
  Digital pheromone system for emergent pathfinding and task routing.
  Minds deposit pheromones on successful paths; others probabilistically follow.
  """
  use GenServer
  require Logger

  defstruct [
    :trails,           # %{path_hash => %{strength: float, deposits: integer}}
    :evaporation_rate,
    :diffusion_rate
  ]

  @evaporation_rate 0.05
  @deposit_amount 0.3

  # Client API
  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def deposit(path_signature, quality \\ 1.0) do
    GenServer.cast(__MODULE__, {:deposit, path_signature, quality})
  end

  def follow(current_state, options) do
    GenServer.call(__MODULE__, {:follow, current_state, options})
  end

  def get_trail_strength(path_signature) do
    GenServer.call(__MODULE__, {:get_strength, path_signature})
  end

  # Server callbacks
  @impl true
  def init(_opts) do
    state = %__MODULE__{
      trails: %{},
      evaporation_rate: @evaporation_rate,
      diffusion_rate: 0.1
    }
    schedule_evaporation()
    {:ok, state}
  end

  @impl true
  def handle_cast({:deposit, signature, quality}, state) do
    path_hash = hash_signature(signature)
    current = Map.get(state.trails, path_hash, %{strength: 0.0, deposits: 0})
    
    new_trail = %{
      strength: min(1.0, current.strength + @deposit_amount * quality),
      deposits: current.deposits + 1,
      last_deposit: DateTime.utc_now()
    }
    
    trails = Map.put(state.trails, path_hash, new_trail)
    {:noreply, %{state | trails: trails}}
  end

  @impl true
  def handle_call({:follow, current_state, options}, _from, state) do
    possible_paths = generate_possible_paths(current_state, options)
    
    # Weighted selection based on pheromone strength
    choice = if Enum.empty?(possible_paths) do
      nil
    else
      weighted_choice(possible_paths, state.trails)
    end
    
    {:reply, choice, state}
  end

  def handle_call({:get_strength, signature}, _from, state) do
    path_hash = hash_signature(signature)
    strength = get_in(state.trails, [path_hash, :strength]) || 0.0
    {:reply, strength, state}
  end

  @impl true
  def handle_info(:evaporate, state) do
    trails = Map.new(state.trails, fn {hash, trail} ->
      new_strength = trail.strength * (1 - @evaporation_rate)
      if new_strength > 0.01 do
        {hash, %{trail | strength: new_strength}}
      else
        nil
      end
    end)
    |> Enum.reject(&is_nil/1)
    |> Map.new()
    
    schedule_evaporation()
    {:noreply, %{state | trails: trails}}
  end

  # Private helpers
  defp hash_signature(signature) do
    :crypto.hash(:sha256, :erlang.term_to_binary(signature)) |> Base.encode16()
  end

  defp generate_possible_paths(_state, _options) do
    # In practice, these would be actions or routing decisions
    [:explore, :exploit, :rest, :broadcast]
  end

  defp weighted_choice(paths, trails) do
    strengths = Enum.map(paths, fn path ->
      trail = Map.get(trails, hash_signature(path), %{strength: 0.01})
      trail.strength
    end)
    
    total = Enum.sum(strengths)
    if total > 0 do
      r = :rand.uniform() * total
      Enum.zip(paths, strengths)
      |> Enum.reduce_while({nil, r}, fn {path, strength}, {_, remaining} ->
        if remaining <= strength do
          {:halt, path}
        else
          {:cont, {nil, remaining - strength}}
        end
      end)
      |> elem(0) || Enum.random(paths)
    else
      Enum.random(paths)
    end
  end

  defp schedule_evaporation do
    Process.send_after(self(), :evaporate, 30_000)
  end
end

defmodule NeuralMeshLite.HyperdimensionalMemory do
  @moduledoc """
  Hyperdimensional associative memory using Vector Symbolic Architectures (VSA).
  Provides high-capacity, noise-robust storage and recall.
  """
  defstruct [
    :memory_vector,    # Superposed memory of all stored patterns
    :item_vectors,     # %{item_id => hypervector}
    :dimensions        # Typically 2048 or 4096
  ]

  @default_dimensions 2048

  def new(dimensions \\ @default_dimensions) do
    %__MODULE__{
      memory_vector: generate_random_vector(dimensions),
      item_vectors: %{},
      dimensions: dimensions
    }
  end

  def store(mem, item_id, data) do
    # Convert data to hypervector and bind to item_id
    data_vector = data_to_vector(data, mem.dimensions)
    item_vector = generate_random_vector(mem.dimensions)
    
    # Binding: item_id ⊗ data_vector
    bound = bind(item_vector, data_vector)
    
    # Add to memory (superposition)
    new_memory = bundle(mem.memory_vector, bound)
    
    item_vectors = Map.put(mem.item_vectors, item_id, item_vector)
    
    %{mem | memory_vector: new_memory, item_vectors: item_vectors}
  end

  def recall(mem, query) do
    query_vector = data_to_vector(query, mem.dimensions)
    
    # Unbind from memory: memory ⊗ query_vector^{-1}
    # Approximates the stored item vector
    recalled = unbind(mem.memory_vector, query_vector)
    
    # Find closest item in item_vectors
    find_closest_item(recalled, mem.item_vectors)
  end

  def associative_recall(mem, partial_query) do
    # Hopfield-like completion: use partial pattern to retrieve full
    partial_vector = data_to_vector(partial_query, mem.dimensions)
    
    # One-step attractor dynamics
    recalled = sign(mem.memory_vector) |> bundle(partial_vector)
    
    find_closest_item(recalled, mem.item_vectors)
  end

  # VSA Operations
  defp generate_random_vector(dim) do
    for _ <- 1..dim, do: Enum.random([-1, 1])
  end

  defp data_to_vector(data, dim) do
    # Hash data to deterministic bipolar vector
    hash = :erlang.phash2(data, 1_000_000)
    :rand.seed(:exsplus, {hash, hash, hash})
    for _ <- 1..dim, do: Enum.random([-1, 1])
  end

  defp bundle(v1, v2) do
    # Element-wise sum then sign (bipolar)
    Enum.zip(v1, v2)
    |> Enum.map(fn {a, b} -> sign(a + b) end)
  end

  defp bind(v1, v2) do
    # Element-wise multiplication (XOR in bipolar)
    Enum.zip(v1, v2)
    |> Enum.map(fn {a, b} -> a * b end)
  end

  defp unbind(bound, v) do
    # Binding is its own inverse in bipolar
    bind(bound, v)
  end

  defp sign(x) when x > 0, do: 1
  defp sign(_), do: -1

  defp find_closest_item(target, item_vectors) do
    if map_size(item_vectors) == 0 do
      nil
    else
      {best_id, _} = Enum.max_by(item_vectors, fn {_, vec} ->
        cosine_similarity(target, vec)
      end)
      best_id
    end
  end

  defp cosine_similarity(v1, v2) do
    dot = Enum.zip(v1, v2) |> Enum.map(fn {a, b} -> a * b end) |> Enum.sum()
    dot / length(v1)
  end
end

    GenServer.call(__MODULE__, :get_knowledge)
  end

  def init(_opts) do

      contribution_count: 0
    }
    {:ok, state}

    # Store experience in collective memory
    collective_memory = HyperdimensionalMemory.store(

      "#{mind_id}_#{state.contribution_count}",
      experience
    )
    
    state = %{state | 
      collective_memory: collective_memory,
      contribution_count: state.contribution_count + 1

    {:noreply, state}

      state.collective_memory,
      query_pattern
    )
    {:reply, result, state}
  end

  def handle_call(:get_knowledge, _from, state) do
    {:reply, state.collective_memory, state}
  end

defmodule NeuralMeshLite.RuleGenome do
  @moduledoc """
  Evolvable behavioral rules encoded as decision trees.
  Mutations can modify conditions, actions, or tree structure.
  """
  defstruct [:rules, :fitness, :generation, :parent_ids]

  @actions [:explore, :exploit, :rest, :broadcast, :evolve]
  @conditions [:energy_high, :fitness_high, :pheromone_strong, :quorum_reached, :task_urgent]

  def random do
    rules = for _ <- 1..Enum.random(3..8) do
      %{
        condition: random_condition(),
        action: Enum.random(@actions),
        priority: :rand.uniform()
      }
    end
    %__MODULE__{
      rules: rules,
      fitness: 0.0,
      generation: 0,
      parent_ids: []
    }
  end

  defp random_condition do
    # Simple condition: e.g., {:energy, :>, 0.5}
    {
      Enum.random(@conditions),
      Enum.random([:>, :<, :==]),
      :rand.uniform()
    }
  end

  def decide(genome, state_vars) do
    # Find highest priority rule whose condition is satisfied
    genome.rules
    |> Enum.sort_by(& &1.priority, :desc)
    |> Enum.find(fn rule -> condition_satisfied?(rule.condition, state_vars) end)
    |> case do
      nil -> :rest
      rule -> rule.action
    end
  end

  defp condition_satisfied?({var, op, threshold}, vars) do
    value = Map.get(vars, var, 0.5)
    case op do
      :> -> value > threshold
      :< -> value < threshold
      :== -> abs(value - threshold) < 0.01
    end
  end

  def mutate(genome, rate \\ 0.1) do
    mutated_rules = Enum.map(genome.rules, fn rule ->
      if :rand.uniform() < rate do
        mutate_rule(rule)
      else
        rule
      end
    end)
    %{genome | rules: mutated_rules, parent_ids: [genome.fitness | genome.parent_ids]}
  end

  defp mutate_rule(rule) do
    case :rand.uniform() do
      x when x < 0.33 -> %{rule | condition: random_condition()}
      x when x < 0.66 -> %{rule | action: Enum.random(@actions)}
      _ -> %{rule | priority: max(0.1, min(1.0, rule.priority + (:rand.uniform() - 0.5) * 0.2))}
    end
  end

  def crossover(g1, g2) do
    # Take rules from both parents
    split = :rand.uniform() * min(length(g1.rules), length(g2.rules)) |> round()
    child_rules = Enum.take(g1.rules, split) ++ Enum.drop(g2.rules, split)
    %__MODULE__{
      rules: child_rules,
      fitness: 0.0,
      generation: max(g1.generation, g2.generation) + 1,
      parent_ids: [g1.fitness, g2.fitness]
    }
  end
end

defmodule NeuralMeshLite.SelfModifier do
  @moduledoc """
  Safe self-modification of the framework's own source code.
  Uses AST mutations with validation in a sandboxed environment.
  """
  require Logger

  @modifiable_modules [
    NeuralMeshLite.MicroMind,
    NeuralMeshLite.EvolutionEngine,
    NeuralMeshLite.SwarmManager
  ]

  def propose_mutation(module) when module in @modifiable_modules do
    source = get_module_source(module)
    ast = Code.string_to_quoted!(source)
    
    # Generate AST mutation
    mutated_ast = mutate_ast(ast)
    mutated_source = Macro.to_string(mutated_ast)
    
    # Validate in sandbox
    if validate_mutation(module, mutated_source) do
      {:ok, mutated_source}
    else
      {:error, :validation_failed}
    end
  end

  def apply_mutation(module, source) do
    path = module_to_path(module)
    backup = path <> ".backup"
    File.copy!(path, backup)
    
    File.write!(path, source)
    
    # Reload module
    Code.compile_file(path)
    
    Logger.info("Applied self-modification to #{module}")
    :ok
  rescue
    e ->
      # Rollback
      File.copy!(backup, path)
      Code.compile_file(path)
      Logger.error("Self-modification failed: #{Exception.message(e)}")
      {:error, e}
  end

  defp get_module_source(module) do
    path = module_to_path(module)
    File.read!(path)
  end

  defp module_to_path(module) do
    module_name = module |> Atom.to_string() |> String.replace("Elixir.", "")
    "lib/" <> Macro.underscore(module_name) <> ".ex"
  end

  defp mutate_ast(ast) do
    # Simplified: randomly change a constant or add a logging statement
    {mutated, _} = Macro.prewalk(ast, false, fn node, changed ->
      if not changed and :rand.uniform() < 0.1 do
        case node do
          {:@, _, [{name, _, [value]}]} when is_number(value) ->
            # Mutate module attribute constant
            new_value = value * (1 + (:rand.uniform() - 0.5) * 0.2)
            {{:@, [], [{name, [], [new_value]}]}, true}
          _ ->
            {node, false}
        end
      else
        {node, changed}
      end
    end)
    mutated
  end

  defp validate_mutation(module, source) do
    # Write to temp file and try to compile
    tmp_path = "/tmp/neural_mesh_lite_test_#{System.monotonic_time()}.ex"
    File.write!(tmp_path, source)
    
    try do
      Code.compile_file(tmp_path)
      true
    rescue
      _ -> false
    after
      File.rm(tmp_path)
    end
  end
end

defmodule NeuralMeshLite.QuorumSensor do
  @moduledoc """
  Tracks collective sentiment for quorum-based decision making.
  """
  use GenServer

  defstruct [:votes, :window_ms, :threshold]

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def cast_vote(proposal_id, mind_id, vote) do
    GenServer.cast(__MODULE__, {:vote, proposal_id, mind_id, vote})
  end

  def check_quorum(proposal_id) do
    GenServer.call(__MODULE__, {:check, proposal_id})
  end

  @impl true
  def init(opts) do
    state = %__MODULE__{
      votes: %{},
      window_ms: Keyword.get(opts, :window_ms, 60_000),
      threshold: Keyword.get(opts, :threshold, 0.6)
    }
    {:ok, state}
  end

  @impl true
  def handle_cast({:vote, proposal_id, mind_id, vote}, state) do
    now = System.monotonic_time(:millisecond)
    votes = Map.update(state.votes, proposal_id, %{}, fn proposal_votes ->
      Map.put(proposal_votes, mind_id, {vote, now})
    end)
    {:noreply, %{state | votes: votes}}
  end

  @impl true
  def handle_call({:check, proposal_id}, _from, state) do
    now = System.monotonic_time(:millisecond)
    proposal_votes = Map.get(state.votes, proposal_id, %{})
    
    # Expire old votes
    active_votes = Enum.filter(proposal_votes, fn {_, {_, ts}} -> 
      now - ts < state.window_ms
    end)
    
    total = length(active_votes)
    approvals = Enum.count(active_votes, fn {_, {v, _}} -> v end)
    
    quorum_reached = if total > 0 do
      approvals / total >= state.threshold
    else
      false
    end
    
    {:reply, %{reached: quorum_reached, approvals: approvals, total: total}, state}
  end
end
