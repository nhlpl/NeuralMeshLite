defmodule NeuralMeshLite.TransposonLibrary do
  @moduledoc """
  Stores and manages successful neural subgraphs (transposons) for reuse across minds.
  """
  use GenServer
  require Logger

  defstruct [:blocks, :max_size, :decay_rate]

  # Client API
  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def extract_block(successful_mind_id, block_signature, block_params, fitness) do
    GenServer.cast(__MODULE__, {:extract, successful_mind_id, block_signature, block_params, fitness})
  end

  def sample_block do
    GenServer.call(__MODULE__, :sample)
  end

  def get_stats do
    GenServer.call(__MODULE__, :get_stats)
  end

  # Server callbacks
  @impl true
  def init(opts) do
    state = %__MODULE__{
      blocks: %{},
      max_size: Keyword.get(opts, :max_size, 200),
      decay_rate: Keyword.get(opts, :decay_rate, 0.05)
    }
    # Periodic decay
    schedule_decay()
    {:ok, state}
  end

  @impl true
  def handle_cast({:extract, mind_id, signature, params, fitness}, state) do
    block_id = :crypto.hash(:sha256, :erlang.term_to_binary(signature)) |> Base.encode16()
    
    blocks = if Map.has_key?(state.blocks, block_id) do
      # Update existing block
      Map.update!(state.blocks, block_id, fn block ->
        %{block | fitness: max(block.fitness, fitness), usage_count: block.usage_count + 1}
      end)
    else
      # New block
      Map.put(state.blocks, block_id, %{
        id: block_id,
        signature: signature,
        params: params,
        fitness: fitness,
        usage_count: 1,
        discovered_at: DateTime.utc_now(),
        source_mind: mind_id
      })
    end
    
    # Prune if over capacity (keep highest fitness)
    blocks = if map_size(blocks) > state.max_size do
      blocks
      |> Enum.sort_by(fn {_, b} -> b.fitness end, :desc)
      |> Enum.take(state.max_size)
      |> Map.new()
    else
      blocks
    end
    
    Logger.debug("Transposon library size: #{map_size(blocks)}")
    {:noreply, %{state | blocks: blocks}}
  end

  @impl true
  def handle_call(:sample, _from, state) do
    if map_size(state.blocks) == 0 do
      {:reply, nil, state}
    else
      # Sample weighted by fitness
      blocks = Map.values(state.blocks)
      total_fitness = Enum.sum(Enum.map(blocks, & &1.fitness))
      r = :rand.uniform() * total_fitness
      
      {selected, _} = Enum.reduce_while(blocks, {nil, r}, fn block, {_, remaining} ->
        if remaining <= block.fitness do
          {:halt, {block, remaining}}
        else
          {:cont, {nil, remaining - block.fitness}}
        end
      end)
      
      {:reply, selected || Enum.random(blocks), state}
    end
  end

  def handle_call(:get_stats, _from, state) do
    stats = %{
      size: map_size(state.blocks),
      avg_fitness: if map_size(state.blocks) > 0 do
        state.blocks |> Map.values() |> Enum.map(& &1.fitness) |> Enum.sum() |> Kernel./(map_size(state.blocks))
      else
        0.0
      end,
      total_usage: state.blocks |> Map.values() |> Enum.map(& &1.usage_count) |> Enum.sum()
    }
    {:reply, stats, state}
  end

  @impl true
  def handle_info(:decay, state) do
    blocks = Map.new(state.blocks, fn {id, block} ->
      decayed_fitness = block.fitness * (1 - state.decay_rate)
      if decayed_fitness > 0.05 do
        {id, %{block | fitness: decayed_fitness}}
      else
        nil
      end
    end)
    |> Enum.reject(&is_nil/1)
    |> Map.new()
    
    schedule_decay()
    {:noreply, %{state | blocks: blocks}}
  end

  defp schedule_decay do
    Process.send_after(self(), :decay, 60_000)  # Every minute
  end
end

defmodule NeuralMeshLite.MicroMind do
  @moduledoc """
  Self-modifying micro-mind with transposon-based architecture evolution,
  epigenetic learning rate modulation, and alternative pathway splicing.
  """
  use GenServer
  require Logger

  alias NeuralMeshLite.{SwarmManager, TransposonLibrary}

  @specialization_config %{
    vision: %{input: 784, output: 10},
    language: %{input: 128, output: 50},
    reasoning: %{input: 64, output: 20},
    general: %{input: 32, output: 10}
  }

  defstruct [
    :id,
    :model,
    :params,
    :specialization,
    :fitness,
    :generation,
    :parent_ids,
    :task_affinity,
    :consolidated_params,
    :performance_history,
    :last_active,
    # New fields
    :epigenetic_tags,        # %{layer_name => %{learning_rate_multiplier: float, ...}}
    :alternative_pathways,   # List of optional sub-networks for splicing
    :active_pathway,         # Currently active pathway index
    :architecture_signature, # For transposon extraction
    :meta_learning_state     # Across-generation learning
  ]

  # Client API (extended)
  def start_link(attrs \\ %{}) do
    GenServer.start_link(__MODULE__, attrs)
  end

  def predict(pid, input) do
    GenServer.call(pid, {:predict, input})
  end

  def get_genome(pid) do
    GenServer.call(pid, :get_genome)
  end

  def update_fitness(pid, fitness, task_type) do
    GenServer.cast(pid, {:update_fitness, fitness, task_type})
  end

  def consolidate(pid) do
    GenServer.cast(pid, :consolidate)
  end

  def evolve_architecture(pid) do
    GenServer.cast(pid, :evolve_architecture)
  end

  def extract_transposon(pid) do
    GenServer.cast(pid, :extract_transposon)
  end

  # Server callbacks
  @impl true
  def init(attrs) do
    id = Map.get(attrs, :id, UUID.uuid4())
    specialization = Map.get(attrs, :specialization, random_specialization())
    
    # Build initial architecture with random topology
    {model, params, architecture_signature, alternative_pathways} = build_dynamic_network(specialization)
    
    state = %__MODULE__{
      id: id,
      model: model,
      params: params,
      specialization: specialization,
      fitness: 0.0,
      generation: Map.get(attrs, :generation, 0),
      parent_ids: Map.get(attrs, :parent_ids, []),
      task_affinity: init_task_affinity(specialization),
      consolidated_params: nil,
      performance_history: [],
      last_active: System.monotonic_time(:second),
      epigenetic_tags: init_epigenetic_tags(model),
      alternative_pathways: alternative_pathways,
      active_pathway: 0,
      architecture_signature: architecture_signature,
      meta_learning_state: %{successful_mutations: [], generation_best: nil}
    }
    
    SwarmManager.register_mind(id, specialization, state.task_affinity)
    {:ok, state}
  end

  @impl true
  def handle_call({:predict, input}, _from, state) do
    tensor = Nx.tensor(input)
    params = state.consolidated_params || state.params
    
    # Apply epigenetic learning rate modulation (affects prediction? No, but tracked for training)
    # For prediction, we can use alternative pathways
    prediction = if length(state.alternative_pathways) > 1 and state.active_pathway < length(state.alternative_pathways) do
      pathway = Enum.at(state.alternative_pathways, state.active_pathway)
      Axon.predict(pathway, params, tensor)
    else
      Axon.predict(state.model, params, tensor)
    end
    
    state = %{state | last_active: System.monotonic_time(:second)}
    {:reply, Nx.to_flat_list(prediction), state}
  end

  def handle_call(:get_genome, _from, state) do
    genome = %{
      specialization: state.specialization,
      params: state.params,
      consolidated_params: state.consolidated_params,
      fitness: state.fitness,
      generation: state.generation,
      parent_ids: state.parent_ids,
      task_affinity: state.task_affinity,
      epigenetic_tags: state.epigenetic_tags,
      architecture_signature: state.architecture_signature,
      alternative_pathways_count: length(state.alternative_pathways)
    }
    {:reply, genome, state}
  end

  @impl true
  def handle_cast({:update_fitness, fitness, task_type}, state) do
    # Epigenetic modulation: adjust learning rates based on performance trend
    epigenetic_tags = if length(state.performance_history) > 5 do
      recent = Enum.take(state.performance_history, 5) |> Enum.map(&elem(&1, 0))
      trend = (List.first(recent) || 0) - (List.last(recent) || 0)
      modulate_epigenetics(state.epigenetic_tags, trend)
    else
      state.epigenetic_tags
    end
    
    affinity_update = if task_type == state.specialization, do: 0.1, else: -0.05
    new_affinity = Map.update!(
      state.task_affinity, 
      task_type, 
      &max(0.1, min(1.0, &1 + affinity_update))
    )
    
    history = [{fitness, task_type} | state.performance_history] |> Enum.take(100)

      Map.update!(state.meta_learning_state, :successful_mutations, &[state.architecture_signature | &1] |> Enum.take(50))
    else
      state.meta_learning_state
    end
    
    # Switch active pathway based on performance
    active_pathway = if length(state.alternative_pathways) > 1 and fitness < 0.5 do
      rem(state.active_pathway + 1, length(state.alternative_pathways))
    else
      state.active_pathway
    end
    
    {:noreply, %{state | 
      fitness: fitness, 
      task_affinity: new_affinity,
      performance_history: history,
      epigenetic_tags: epigenetic_tags,
      meta_learning_state: meta,
      active_pathway: active_pathway
    }}
  end

  def handle_cast(:consolidate, state) do
    if state.fitness > 0.7 and length(state.performance_history) > 10 do
      consolidated = if state.consolidated_params do
        deep_merge(state.consolidated_params, state.params, fn old, new ->
          Nx.add(Nx.multiply(old, 0.85), Nx.multiply(new, 0.15))
        end)
      else
        state.params
      end
      
      # Extract successful architecture as transposon
      if state.fitness > 0.85 do
        TransposonLibrary.extract_block(
          state.id,
          state.architecture_signature,
          state.params,
          state.fitness
        )
      end
      
      {:noreply, %{state | consolidated_params: consolidated}}
    else
      {:noreply, state}
    end
  end

  def handle_cast(:evolve_architecture, state) do
    # Transposon-based architecture mutation
    case TransposonLibrary.sample_block() do
      nil -> 
        # Random mutation if no transposon available
        new_state = mutate_architecture_randomly(state)
        {:noreply, new_state}
      block ->
        # Insert transposon block into architecture
        new_state = insert_transposon(state, block)
        Logger.debug("Mind #{state.id} evolved via transposon #{block.id}")
        {:noreply, new_state}
    end
  end

  def handle_cast(:extract_transposon, state) do
    if state.fitness > 0.8 do
      TransposonLibrary.extract_block(
        state.id,
        state.architecture_signature,
        state.params,
        state.fitness
      )
    end
    {:noreply, state}
  end

  # Private helpers
  defp random_specialization do
    Enum.random([:vision, :language, :reasoning, :general])
  end

  defp build_dynamic_network(specialization) do
    config = @specialization_config[specialization]
    
    # Evolved: variable number of hidden layers (1-3) with random widths
    num_hidden = Enum.random(1..3)
    hidden_sizes = for _ <- 1..num_hidden, do: Enum.random([16, 24, 32, 48, 64])
    
    # Alternative pathway: a parallel subnetwork
    has_alternative = Enum.random([true, false])
    
    model = 
      Axon.input("input", shape: {nil, config.input})
      |> then(fn ax ->
        Enum.reduce(hidden_sizes, ax, fn size, acc ->
          acc |> Axon.dense(size, activation: :relu) |> Axon.dropout(rate: 0.1)
        end)
      end)
      |> Axon.dense(config.output, activation: :softmax)
    
    alternative_pathways = if has_alternative do
      alt = Axon.input("input", shape: {nil, config.input})
            |> Axon.dense(32, activation: :tanh)
            |> Axon.dense(config.output, activation: :softmax)
      [alt]
    else
      []
    end
    
    signature = %{
      specialization: specialization,
      hidden_sizes: hidden_sizes,
      has_alternative: has_alternative
    }
    
    {model, Axon.init(model), signature, alternative_pathways}
  end

  defp init_epigenetic_tags(model) do
    # Assign initial learning rate multipliers per layer
    %{
      learning_rate_multipliers: %{default: 1.0},
      plasticity: 0.5
    }
  end

  defp init_task_affinity(specialization) do
    base = %{vision: 0.5, language: 0.5, reasoning: 0.5, general: 0.5}
    Map.put(base, specialization, 0.8)
  end

  defp modulate_epigenetics(tags, performance_trend) do
    # Increase plasticity if improving, decrease if stagnating
    new_plasticity = max(0.1, min(1.0, tags.plasticity + performance_trend * 0.1))
    multipliers = Map.new(tags.learning_rate_multipliers, fn {k, v} -> 
      {k, v * (1.0 + (new_plasticity - 0.5) * 0.2)}
    end)
    %{tags | plasticity: new_plasticity, learning_rate_multipliers: multipliers}
  end

  defp mutate_architecture_randomly(state) do
    # Add/remove layer, change activation, etc.
    # Simplified: just rebuild with new random topology
    {model, params, signature, pathways} = build_dynamic_network(state.specialization)
    %{state | model: model, params: params, architecture_signature: signature, alternative_pathways: pathways}
  end

  defp insert_transposon(state, block) do
    # Insert the transposon's architecture pattern
    new_sig = block.signature
    {model, params, signature, pathways} = build_dynamic_network(new_sig.specialization)
    # Merge transposon parameters with existing (where compatible)
    merged_params = deep_merge(state.params, block.params, fn p1, p2 -> p2 end)
    %{state | model: model, params: merged_params, architecture_signature: signature, alternative_pathways: pathways}
  end

  defp deep_merge(map1, map2, fun) when is_map(map1) and is_map(map2) do
    Map.merge(map1, map2, fn _k, v1, v2 -> 
      if is_map(v1) and is_map(v2) do
        deep_merge(v1, v2, fun)
      else
        fun.(v1, v2)
      end
    end)
  end
  defp deep_merge(_, other, fun), do: fun.(other, other)
end

defmodule NeuralMeshLite.MetaLearner do
  @moduledoc """
  Meta-learning across generations: learns optimal mutation strategies,
  initializations, and hyperparameters from evolutionary history.
  """
  use GenServer
  require Logger

  defstruct [
    :history,
    :learned_mutation_rates,
    :learned_init_params,
    :successful_architectures
  ]

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def record_generation(stats, best_genome) do
    GenServer.cast(__MODULE__, {:record, stats, best_genome})
  end

  def get_learned_mutation_rate do
    GenServer.call(__MODULE__, :get_mutation_rate)
  end

  def get_learned_init_params(specialization) do
    GenServer.call(__MODULE__, {:get_init_params, specialization})
  end

  @impl true
  def init(_opts) do
    state = %__MODULE__{
      history: [],
      learned_mutation_rates: %{default: 0.15},
      learned_init_params: %{},
      successful_architectures: []
    }
    {:ok, state}
  end

  @impl true
  def handle_cast({:record, stats, best_genome}, state) do
    history = [%{generation: stats.generation, best_fitness: best_genome.fitness, 
                 specialization: best_genome.specialization} | state.history] |> Enum.take(500)
    
    # Learn optimal mutation rate from improvement rate
    if length(history) > 10 do
      improvements = history
      |> Enum.chunk_every(2, 1, :discard)
      |> Enum.map(fn [curr, prev] -> curr.best_fitness - prev.best_fitness end)
      
      avg_improvement = Enum.sum(improvements) / length(improvements)
      # Adjust mutation rate: increase if plateaued, decrease if volatile
      current_rate = state.learned_mutation_rates.default
      new_rate = cond do
        avg_improvement < 0.01 -> min(0.4, current_rate * 1.2)
        avg_improvement > 0.05 -> max(0.05, current_rate * 0.9)
        true -> current_rate
      end
      
      learned_mutation_rates = Map.put(state.learned_mutation_rates, :default, new_rate)
      
      # Store successful architecture signatures
      if best_genome.fitness > 0.85 do
        archs = [best_genome.architecture_signature | state.successful_architectures] |> Enum.uniq() |> Enum.take(50)
        state = %{state | successful_architectures: archs}
      end
      
      {:noreply, %{state | history: history, learned_mutation_rates: learned_mutation_rates}}
    else
      {:noreply, %{state | history: history}}
    end
  end

  @impl true
  def handle_call(:get_mutation_rate, _from, state) do
    {:reply, state.learned_mutation_rates.default, state}
  end

  def handle_call({:get_init_params, specialization}, _from, state) do
    # Return learned good initialization for this specialization
    params = Map.get(state.learned_init_params, specialization, %{})
    {:reply, params, state}
  end
end

defmodule NeuralMeshLite.ReasoningPlanner do
  @moduledoc """
  Integrates DeepSeek reasoning for complex task decomposition and strategy planning.
  """
  require Logger

  @deepseek_api "https://api.deepseek.com/v1/chat/completions"

  def plan_complex_task(task_description, available_minds) do
    api_key = System.get_env("DEEPSEEK_API_KEY")
    if !api_key do
      Logger.warn("DeepSeek API key not set; using fallback planning")
      fallback_plan(task_description, available_minds)
    else
      prompt = build_planning_prompt(task_description, available_minds)
      
      body = %{
        model: "deepseek-reasoner",
        messages: [%{role: "user", content: prompt}],
        temperature: 0.2,
        max_tokens: 500
      }
      
      headers = [
        {"Authorization", "Bearer #{api_key}"},
        {"Content-Type", "application/json"}
      ]
      
      case Req.post(@deepseek_api, json: body, headers: headers) do
        {:ok, %{status: 200, body: %{"choices" => [%{"message" => %{"content" => content}}]}}} ->
          parse_plan(content)
        _ ->
          fallback_plan(task_description, available_minds)
      end
    end
  end

  defp build_planning_prompt(task, minds) do
    mind_summary = Enum.map(minds, fn m -> 
      "- #{m.specialization} (fitness: #{Float.round(m.fitness, 2)}, affinity: #{inspect(m.affinity)})"
    end) |> Enum.join("\n")
    
    """
    You are planning a multi-step task for a swarm of specialized neural networks.
    
    Task: #{task}
    
    Available minds:
    #{mind_summary}
    
    Propose a decomposition of the task into subtasks, and assign each subtask to the most appropriate mind specialization.
    Return a JSON list of steps, each with "subtask" and "specialization".
    """
  end

  defp parse_plan(response) do
    case Jason.decode(response) do
      {:ok, plan} when is_list(plan) -> plan
      _ -> []
    end
  end

  defp fallback_plan(task, minds) do
    # Simple heuristic: assign to highest affinity mind
    specs = Enum.map(minds, & &1.specialization) |> Enum.uniq()
    [%{"subtask" => task, "specialization" => Enum.random(specs)}]
  end
end
