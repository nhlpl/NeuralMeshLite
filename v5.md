```elixir
defmodule NeuralMeshLite.PredictiveCoding do
  @moduledoc """
  Hierarchical predictive coding for temporal learning.
  Each layer predicts the activity of the layer below; prediction errors
  propagate upward and update internal models.
  """
  defstruct [
    :levels,           # List of predictive layers
    :predictions,      # Current predictions at each level
    :errors,           # Prediction errors
    :learning_rate
  ]

  def new(num_levels \\ 3, input_size \\ 64) do
    levels = for i <- 1..num_levels do
      size = div(input_size, :math.pow(2, num_levels - i) |> round())
      %{
        weights: random_matrix(size, size),
        bias: random_vector(size),
        state: random_vector(size)
      }
    end
    
    %__MODULE__{
      levels: levels,
      predictions: [],
      errors: [],
      learning_rate: 0.01
    }
  end

  def forward(pc, input) do
    # Bottom-up pass: generate predictions at each level
    {levels, predictions, errors} = Enum.reduce(
      Enum.with_index(pc.levels),
      {[], [], []},
      fn {level, idx}, {acc_levels, acc_preds, acc_errs} ->
        # Prediction from level above (or input for bottom)
        input_signal = if idx == 0, do: input, else: Enum.at(acc_preds, idx - 1)
        
        # Generate prediction
        prediction = level.weights |> Nx.dot(input_signal) |> Nx.add(level.bias) |> Nx.tanh()
        
        # Compute error (for top-down correction)
        error = Nx.subtract(input_signal, prediction)
        
        # Update level state
        updated_level = %{level | state: prediction}
        
        {[updated_level | acc_levels], [prediction | acc_preds], [error | acc_errs]}
      end
    )
    
    %{pc | levels: Enum.reverse(levels), predictions: Enum.reverse(predictions), errors: Enum.reverse(errors)}
  end

  def learn(pc) do
    # Update weights based on prediction errors
    levels = Enum.map(Enum.with_index(pc.levels), fn {level, idx} ->
      error = Enum.at(pc.errors, idx)
      input = if idx == 0, do: nil, else: Enum.at(pc.predictions, idx - 1)
      
      if input do
        # Gradient descent on weights
        gradient = Nx.outer(error, input)
        new_weights = Nx.subtract(level.weights, Nx.multiply(gradient, pc.learning_rate))
        new_bias = Nx.subtract(level.bias, Nx.multiply(error, pc.learning_rate))
        %{level | weights: new_weights, bias: new_bias}
      else
        level
      end
    end)
    
    %{pc | levels: levels}
  end

  defp random_matrix(rows, cols) do
    for _ <- 1..rows, do: for(_ <- 1..cols, do: :rand.uniform() * 2 - 1)
  end

  defp random_vector(size) do
    for _ <- 1..size, do: :rand.uniform() * 2 - 1
  end
end
```

```elixir
defmodule NeuralMeshLite.QuantumSuperposition do
  @moduledoc """
  Quantum-inspired superposition of multiple strategies.
  Minds maintain a probability amplitude distribution over possible actions;
  measurement collapses to a single action, but learning updates all amplitudes.
  """
  defstruct [
    :amplitudes,       # %{action => complex amplitude}
    :phase,            # Global phase
    :entanglement      # Correlations with other minds (simplified)
  ]

  @actions [:explore, :exploit, :rest, :broadcast, :evolve, :defend, :reproduce]

  def new do
    # Initialize with equal amplitudes
    n = length(@actions)
    amp = :math.sqrt(1.0 / n)
    amplitudes = Map.new(@actions, fn action -> {action, {amp, 0.0}} end)
    
    %__MODULE__{
      amplitudes: amplitudes,
      phase: 0.0,
      entanglement: %{}
    }
  end

  def measure(qs) do
    # Collapse superposition to a single action based on probability = |amplitude|^2
    probs = Map.new(qs.amplitudes, fn {action, {re, im}} ->
      {action, re * re + im * im}
    end)
    
    total = Enum.sum(Map.values(probs))
    if total == 0 do
      Enum.random(@actions)
    else
      r = :rand.uniform() * total
      {action, _} = Enum.reduce_while(probs, {nil, r}, fn {a, p}, {_, rem} ->
        if rem <= p, do: {:halt, {a, rem}}, else: {:cont, {nil, rem - p}}
      end)
      action || Enum.random(@actions)
    end
  end

  def apply_gate(qs, action, reward) do
    # Amplify amplitude of successful actions (quantum reinforcement)
    # U = exp(i * reward * |action><action|)
    phase_shift = reward * :math.pi() / 4
    
    new_amplitudes = Map.new(qs.amplitudes, fn {a, {re, im}} ->
      if a == action do
        # Apply phase rotation: e^(iθ) * (re + i*im)
        new_re = re * :math.cos(phase_shift) - im * :math.sin(phase_shift)
        new_im = re * :math.sin(phase_shift) + im * :math.cos(phase_shift)
        {a, {new_re, new_im}}
      else
        {a, {re, im}}
      end
    end)
    
    # Normalize
    total_prob = new_amplitudes |> Map.values() |> Enum.map(fn {re, im} -> re*re + im*im end) |> Enum.sum()
    norm_factor = :math.sqrt(total_prob)
    
    normalized = if norm_factor > 0 do
      Map.new(new_amplitudes, fn {a, {re, im}} ->
        {a, {re / norm_factor, im / norm_factor}}
      end)
    else
      new_amplitudes
    end
    
    %{qs | amplitudes: normalized}
  end

  def entangle(qs1, qs2, strength \\ 0.5) do
    # Simplified entanglement: correlate amplitude updates
    entanglement_id = UUID.uuid4()
    qs1 = %{qs1 | entanglement: Map.put(qs1.entanglement, entanglement_id, {qs2, strength})}
    qs2 = %{qs2 | entanglement: Map.put(qs2.entanglement, entanglement_id, {qs1, strength})}
    {qs1, qs2}
  end

  def interfere(qs1, qs2) do
    # Superpose two minds' strategies (quantum interference)
    combined_amps = Map.merge(qs1.amplitudes, qs2.amplitudes, fn _a, {re1, im1}, {re2, im2} ->
      {re1 + re2, im1 + im2}
    end)
    
    # Normalize
    total = combined_amps |> Map.values() |> Enum.map(fn {re, im} -> re*re + im*im end) |> Enum.sum()
    norm = :math.sqrt(total)
    
    normalized = if norm > 0 do
      Map.new(combined_amps, fn {a, {re, im}} -> {a, {re / norm, im / norm}} end)
    else
      combined_amps
    end
    
    %__MODULE__{amplitudes: normalized, phase: (qs1.phase + qs2.phase) / 2, entanglement: %{}}
  end
end
```

```elixir
defmodule NeuralMeshLite.FractalHierarchy do
  @moduledoc """
  Multi-scale nested organization: minds contain subminds recursively.
  Enables hierarchical planning and abstraction.
  """
  defstruct [
    :mind_id,
    :subminds,         # List of child minds
    :parent_mind,      # PID of parent (nil for root)
    :abstraction_level # 0 = leaf, higher = more abstract
  ]

  def create_hierarchy(depth \\ 3, branching_factor \\ 3) do
    create_level(nil, depth, branching_factor, 0)
  end

  defp create_level(parent, depth, branching, current_depth) do
    mind_id = UUID.uuid4()
    
    subminds = if current_depth < depth - 1 do
      for _ <- 1..branching do
        create_level(mind_id, depth, branching, current_depth + 1)
      end
    else
      []
    end
    
    %__MODULE__{
      mind_id: mind_id,
      subminds: subminds,
      parent_mind: parent,
      abstraction_level: depth - current_depth - 1
    }
  end

  def propagate_up(hierarchy, leaf_data) do
    # Bottom-up: aggregate from leaves to root
    if hierarchy.subminds == [] do
      # Leaf: process data directly
      {hierarchy, leaf_data}
    else
      # Internal node: aggregate from children
      {updated_children, child_results} = hierarchy.subminds
      |> Enum.map(&propagate_up(&1, leaf_data))
      |> Enum.unzip()
      
      # Aggregate (e.g., average, max, or learned combination)
      aggregated = aggregate_results(child_results)
      
      {%{hierarchy | subminds: updated_children}, aggregated}
    end
  end

  def propagate_down(hierarchy, root_signal) do
    # Top-down: distribute signal to children
    if hierarchy.subminds == [] do
      {hierarchy, root_signal}
    else
      # Distribute signal to each child (with possible modulation)
      updated_children = Enum.map(hierarchy.subminds, fn child ->
        modulated_signal = modulate_signal(root_signal, child.abstraction_level)
        {updated_child, _} = propagate_down(child, modulated_signal)
        updated_child
      end)
      
      {%{hierarchy | subminds: updated_children}, root_signal}
    end
  end

  defp aggregate_results(results) do
    # Simple averaging; can be replaced with learned aggregation
    if Enum.empty?(results), do: nil, else: Enum.reduce(results, &add_vectors/2)
  end

  defp add_vectors(v1, v2) when is_list(v1) and is_list(v2) do
    Enum.zip(v1, v2) |> Enum.map(fn {a, b} -> a + b end)
  end
  defp add_vectors(v1, v2), do: v1

  defp modulate_signal(signal, level) when is_list(signal) do
    # Higher levels get more abstract/compressed signal
    compression = 1.0 / :math.pow(2, level)
    Enum.map(signal, &(&1 * compression))
  end
  defp modulate_signal(signal, _), do: signal
end
```

```elixir
defmodule NeuralMeshLite.SymbiosisEngine do
  @moduledoc """
  Manages symbiotic relationships between minds:
  - Mutualism (both benefit)
  - Commensalism (one benefits, other neutral)
  - Parasitism (one benefits, other harmed)
  """
  use GenServer
  require Logger

  defstruct [
    :relationships,    # %{{mind1, mind2} => type}
    :fitness_shares,   # %{mind_id => %{partner_id => share_percentage}}
    :resource_transfers
  ]

  @types [:mutualism, :commensalism, :parasitism, :neutral]

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def form_relationship(mind1, mind2, type) when type in @types do
    GenServer.cast(__MODULE__, {:form, mind1, mind2, type})
  end

  def dissolve_relationship(mind1, mind2) do
    GenServer.cast(__MODULE__, {:dissolve, mind1, mind2})
  end

  def transfer_fitness(donor, recipient, amount) do
    GenServer.cast(__MODULE__, {:transfer, donor, recipient, amount})
  end

  def get_partner_fitness_boost(mind_id) do
    GenServer.call(__MODULE__, {:get_boost, mind_id})
  end

  @impl true
  def init(_opts) do
    state = %__MODULE__{
      relationships: %{},
      fitness_shares: %{},
      resource_transfers: %{}
    }
    {:ok, state}
  end

  @impl true
  def handle_cast({:form, mind1, mind2, type}, state) do
    key = {min(mind1, mind2), max(mind1, mind2)}
    relationships = Map.put(state.relationships, key, type)
    
    # Setup fitness sharing based on type
    fitness_shares = case type do
      :mutualism ->
        state.fitness_shares
        |> Map.update(mind1, %{mind2 => 0.2}, &Map.put(&1, mind2, 0.2))
        |> Map.update(mind2, %{mind1 => 0.2}, &Map.put(&1, mind1, 0.2))
      :commensalism ->
        state.fitness_shares
        |> Map.update(mind2, %{mind1 => 0.1}, &Map.put(&1, mind1, 0.1))
      :parasitism ->
        state.fitness_shares
        |> Map.update(mind2, %{mind1 => 0.3}, &Map.put(&1, mind1, 0.3))
      :neutral ->
        state.fitness_shares
    end
    
    Logger.debug("Symbiosis formed: #{mind1} <-> #{mind2} (#{type})")
    {:noreply, %{state | relationships: relationships, fitness_shares: fitness_shares}}
  end

  def handle_cast({:dissolve, mind1, mind2}, state) do
    key = {min(mind1, mind2), max(mind1, mind2)}
    relationships = Map.delete(state.relationships, key)
    
    fitness_shares = state.fitness_shares
    |> Map.update(mind1, %{}, &Map.delete(&1, mind2))
    |> Map.update(mind2, %{}, &Map.delete(&1, mind1))
    
    {:noreply, %{state | relationships: relationships, fitness_shares: fitness_shares}}
  end

  def handle_cast({:transfer, donor, recipient, amount}, state) do
    transfers = Map.update(
      state.resource_transfers,
      {donor, recipient},
      amount,
      &(&1 + amount)
    )
    {:noreply, %{state | resource_transfers: transfers}}
  end

  @impl true
  def handle_call({:get_boost, mind_id}, _from, state) do
    shares = Map.get(state.fitness_shares, mind_id, %{})
    total_boost = Enum.sum(Map.values(shares))
    {:reply, total_boost, state}
  end
end
```

```elixir
defmodule NeuralMeshLite.MemePool do
  @moduledoc """
  Cultural transmission of behaviors (memes) between minds.
  Memes are learned patterns that can spread horizontally.
  """
  use GenServer
  require Logger

  defstruct [
    :memes,            # %{meme_id => %{content, fitness, spread_count}}
    :meme_genomes      # %{meme_id => behavioral rules}
  ]

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def create_meme(creator_id, behavior_pattern, fitness) do
    GenServer.cast(__MODULE__, {:create, creator_id, behavior_pattern, fitness})
  end

  def spread_meme(meme_id, target_id) do
    GenServer.cast(__MODULE__, {:spread, meme_id, target_id})
  end

  def sample_meme do
    GenServer.call(__MODULE__, :sample)
  end

  def get_trending(n \\ 5) do
    GenServer.call(__MODULE__, {:trending, n})
  end

  @impl true
  def init(_opts) do
    state = %__MODULE__{
      memes: %{},
      meme_genomes: %{}
    }
    schedule_decay()
    {:ok, state}
  end

  @impl true
  def handle_cast({:create, creator_id, pattern, fitness}, state) do
    meme_id = UUID.uuid4()
    meme = %{
      id: meme_id,
      creator: creator_id,
      content: pattern,
      fitness: fitness,
      spread_count: 0,
      created_at: DateTime.utc_now()
    }
    
    memes = Map.put(state.memes, meme_id, meme)
    meme_genomes = Map.put(state.meme_genomes, meme_id, pattern)
    
    Logger.debug("Meme created: #{meme_id} by #{creator_id}")
    {:noreply, %{state | memes: memes, meme_genomes: meme_genomes}}
  end

  def handle_cast({:spread, meme_id, target_id}, state) do
    if Map.has_key?(state.memes, meme_id) do
      memes = Map.update!(state.memes, meme_id, fn meme ->
        %{meme | spread_count: meme.spread_count + 1}
      end)
      
      # Notify target mind to adopt meme (handled by SwarmManager)
      send(target_id, {:adopt_meme, state.meme_genomes[meme_id]})
      
      {:noreply, %{state | memes: memes}}
    else
      {:noreply, state}
    end
  end

  @impl true
  def handle_call(:sample, _from, state) do
    if map_size(state.memes) == 0 do
      {:reply, nil, state}
    else
      # Sample weighted by fitness * spread_count
      memes = Map.values(state.memes)
      weights = Enum.map(memes, fn m -> m.fitness * (1 + :math.log(m.spread_count + 1)) end)
      total = Enum.sum(weights)
      
      if total > 0 do
        r = :rand.uniform() * total
        {selected, _} = Enum.reduce_while(Enum.zip(memes, weights), {nil, r}, fn {m, w}, {_, rem} ->
          if rem <= w, do: {:halt, {m, rem}}, else: {:cont, {nil, rem - w}}
        end)
        {:reply, selected || Enum.random(memes), state}
      else
        {:reply, Enum.random(memes), state}
      end
    end
  end

  def handle_call({:trending, n}, _from, state) do
    trending = state.memes
    |> Map.values()
    |> Enum.sort_by(&(&1.spread_count), :desc)
    |> Enum.take(n)
    {:reply, trending, state}
  end

  @impl true
  def handle_info(:decay, state) do
    # Memes lose fitness over time (forgetting)
    memes = Map.new(state.memes, fn {id, meme} ->
      decayed_fitness = meme.fitness * 0.99
      if decayed_fitness > 0.1 do
        {id, %{meme | fitness: decayed_fitness}}
      else
        nil
      end
    end)
    |> Enum.reject(&is_nil/1)
    |> Map.new()
    
    schedule_decay()
    {:noreply, %{state | memes: memes}}
  end

  defp schedule_decay do
    Process.send_after(self(), :decay, 120_000)
  end
end
```

```elixir
defmodule NeuralMeshLite.Homeostasis do
  @moduledoc """
  Predictive allostatic regulation of internal states.
  Minds maintain energy, stress, and confidence levels within optimal ranges.
  """
  defstruct [
    :energy,           # 0-1
    :stress,           # 0-1
    :confidence,       # 0-1
    :setpoints,        # %{energy: target, stress: target, confidence: target}
    :prediction_model  # Predicts future states
  ]

  def new do
    %__MODULE__{
      energy: 1.0,
      stress: 0.3,
      confidence: 0.5,
      setpoints: %{energy: 0.8, stress: 0.2, confidence: 0.7},
      prediction_model: %{weights: random_matrix(3, 3), bias: random_vector(3)}
    }
  end

  def update(homeo, action, outcome) do
    # Update internal states based on action and outcome
    energy_cost = action_cost(action)
    stress_change = stress_impact(action, outcome)
    confidence_change = confidence_impact(outcome)
    
    new_energy = max(0.1, min(1.0, homeo.energy - energy_cost))
    new_stress = max(0.1, min(1.0, homeo.stress + stress_change))
    new_confidence = max(0.1, min(1.0, homeo.confidence + confidence_change))
    
    %{homeo | energy: new_energy, stress: new_stress, confidence: new_confidence}
  end

  def predict_needs(homeo) do
    # Predict future state deviation from setpoints
    current = [homeo.energy, homeo.stress, homeo.confidence]
    predicted = current |> Nx.tensor() |> Nx.dot(homeo.prediction_model.weights) |> Nx.add(homeo.prediction_model.bias)
    
    setpoints = [homeo.setpoints.energy, homeo.setpoints.stress, homeo.setpoints.confidence]
    deviations = Enum.zip(predicted, setpoints) |> Enum.map(fn {p, s} -> p - s end)
    
    # Determine most urgent need
    {need, _} = [energy: Enum.at(deviations, 0), stress: Enum.at(deviations, 1), confidence: Enum.at(deviations, 2)]
    |> Enum.max_by(fn {_, d} -> abs(d) end)
    
    need
  end

  def allostatic_action(homeo) do
    need = predict_needs(homeo)
    case need do
      :energy -> :rest
      :stress -> :explore   # Reduce stress through novelty
      :confidence -> :exploit  # Build confidence through success
    end
  end

  defp action_cost(:explore), do: 0.05
  defp action_cost(:exploit), do: 0.03
  defp action_cost(:rest), do: -0.1
  defp action_cost(_), do: 0.02

  defp stress_impact(:explore, :success), do: -0.05
  defp stress_impact(_, :failure), do: 0.1
  defp stress_impact(_, _), do: 0.0

  defp confidence_impact(:success), do: 0.05
  defp confidence_impact(:failure), do: -0.1
  defp confidence_impact(_), do: 0.0

  defp random_matrix(r, c), do: for(_ <- 1..r, do: for(_ <- 1..c, do: :rand.uniform()))
  defp random_vector(s), do: for(_ <- 1..s, do: :rand.uniform())
end
```

```elixir
defmodule NeuralMeshLite.EmergentLanguage do
  @moduledoc """
  Emergent symbolic communication protocol between minds.
  Minds negotiate shared tokens for concepts through repeated interaction.
  """
  use GenServer
  require Logger

  defstruct [
    :lexicon,          # %{concept => token}
    :reverse_lexicon,  # %{token => concept}
    :negotiation_history,
    :usage_stats       # Track token usage for pruning
  ]

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  def propose_token(mind_id, concept, token) do
    GenServer.cast(__MODULE__, {:propose, mind_id, concept, token})
  end

  def express(mind_id, concept) do
    GenServer.call(__MODULE__, {:express, mind_id, concept})
  end

  def interpret(mind_id, token) do
    GenServer.call(__MODULE__, {:interpret, mind_id, token})
  end

  def get_shared_lexicon do
    GenServer.call(__MODULE__, :get_lexicon)
  end

  @impl true
  def init(_opts) do
    state = %__MODULE__{
      lexicon: %{},
      reverse_lexicon: %{},
      negotiation_history: [],
      usage_stats: %{}
    }
    {:ok, state}
  end

  @impl true
  def handle_cast({:propose, mind_id, concept, token}, state) do
    # Simple consensus: if token not used, adopt it
    if !Map.has_key?(state.reverse_lexicon, token) do
      lexicon = Map.put(state.lexicon, concept, token)
      reverse = Map.put(state.reverse_lexicon, token, concept)
      usage = Map.put(state.usage_stats, token, %{uses: 0, proposer: mind_id})
      
      Logger.debug("New token adopted: '#{token}' for concept '#{concept}' by #{mind_id}")
      {:noreply, %{state | lexicon: lexicon, reverse_lexicon: reverse, usage_stats: usage}}
    else
      # Conflict: record for future resolution
      history = [{mind_id, concept, token, DateTime.utc_now()} | state.negotiation_history]
      {:noreply, %{state | negotiation_history: history}}
    end
  end

  @impl true
  def handle_call({:express, mind_id, concept}, _from, state) do
    token = Map.get(state.lexicon, concept)
    if token do
      # Update usage stats
      usage_stats = Map.update!(state.usage_stats, token, &%{&1 | uses: &1.uses + 1})
      state = %{state | usage_stats: usage_stats}
    end
    {:reply, token, state}
  end

  def handle_call({:interpret, _mind_id, token}, _from, state) do
    concept = Map.get(state.reverse_lexicon, token)
    {:reply, concept, state}
  end
```

```elixir
defmodule NeuralMeshLite.TemporalMemory do
  @moduledoc """
  HTM-inspired temporal sequence memory.
  Learns transitions between patterns and predicts next states.
  """
  defstruct [
    :columns,          # List of mini-columns
    :cells,            # 2D array of cells (columns x cells_per_column)
    :synapses,         # %{{from_col, from_cell} => %{{to_col, to_cell} => permanence}}
    :active_cells,     # Currently active cells
    :predicted_cells,  # Cells predicted to become active
    :learning_enabled
  ]

  def new(num_columns \\ 2048, cells_per_column \\ 32) do
    %__MODULE__{
      columns: for(_ <- 1..num_columns, do: %{overlap: 0, active: false}),
      cells: for(_ <- 1..num_columns, do: for(_ <- 1..cells_per_column, do: %{active: false, predicted: false})),
      synapses: %{},
      active_cells: [],
      predicted_cells: [],
      learning_enabled: true
    }
  end

  def compute(tm, input_sdr) do
    # SDR = Sparse Distributed Representation (list of active column indices)
    
    # 1. Activate columns based on input
    columns = Enum.map(Enum.with_index(tm.columns), fn {col, idx} ->
      overlap = if idx in input_sdr, do: 1, else: 0
      active = overlap > 0
      %{col | overlap: overlap, active: active}
    end)
    
    # 2. Activate cells: for each active column, choose cell that was predicted (if any)
    {cells, active_cells, predicted_cells} = Enum.reduce(
      Enum.with_index(columns),
      {tm.cells, [], []},
      fn {col, col_idx}, {acc_cells, acc_active, acc_pred} ->
        if col.active do
          # Bursting: if no predicted cell, activate all cells
          col_cells = Enum.at(acc_cells, col_idx)
          predicted_in_col = Enum.filter(Enum.with_index(col_cells), fn {cell, _} -> cell.predicted end)
          
          if predicted_in_col != [] do
            # Activate only predicted cells
            {cell_idx, _} = Enum.random(predicted_in_col)
            updated_col_cells = List.update_at(col_cells, cell_idx, &%{&1 | active: true})
            {
              List.replace_at(acc_cells, col_idx, updated_col_cells),
              [{col_idx, cell_idx} | acc_active],
              acc_pred
            }
          else
            # Burst: activate all cells
            updated_col_cells = Enum.map(col_cells, &%{&1 | active: true})
            new_active = Enum.map(0..(length(col_cells)-1), &{col_idx, &1})
            {
              List.replace_at(acc_cells, col_idx, updated_col_cells),
              new_active ++ acc_active,
              acc_pred
            }
          end
        else
          {acc_cells, acc_active, acc_pred}
        end
      end
    )
    
    # 3. Make predictions for next step
    {cells, predicted_cells} = predict_next(tm, cells, active_cells)
    
    # 4. Learn if enabled
    synapses = if tm.learning_enabled do
      learn_transitions(tm.synapses, tm.active_cells, active_cells)
    else
      tm.synapses
    end
    
    %{tm | columns: columns, cells: cells, synapses: synapses, 
          active_cells: active_cells, predicted_cells: predicted_cells}
  end

  defp predict_next(tm, cells, active_cells) do
    # For each active cell, activate its distal synapses to predict next cells
    predictions = Enum.flat_map(active_cells, fn {col, cell} ->
      Map.get(tm.synapses, {col, cell}, %{})
      |> Enum.filter(fn {_, perm} -> perm > 0.5 end)
      |> Enum.map(fn {{next_col, next_cell}, _} -> {next_col, next_cell} end)
    end)
    |> Enum.uniq()
    
    # Mark predicted cells
    cells = Enum.reduce(predictions, cells, fn {col, cell}, acc ->
      List.update_at(acc, col, fn col_cells ->
        List.update_at(col_cells, cell, &%{&1 | predicted: true})
      end)
    end)
    
    {cells, predictions}
  end

  defp learn_transitions(synapses, prev_active, curr_active) do
    # Strengthen synapses from previously active to currently active
    Enum.reduce(prev_active, synapses, fn from, acc ->
      Enum.reduce(curr_active, acc, fn to, acc2 ->
        key = {from, to}
        current = Map.get(acc2, key, 0.0)
        new_perm = min(1.0, current + 0.1)
        Map.put(acc2, key, new_perm)
      end)
    end)
  end
end
```
