defmodule NeuralMeshLite.Spacetime do
  @moduledoc """
  Relativistic spacetime cognition using Minkowski metric.
  Agents perceive events in 4D spacetime and optimize proper time.
  """
  defstruct [
    :worldline,        # List of {t, x, y, z} events
    :velocity,         # {vx, vy, vz} as fraction of c
    :gamma,            # Lorentz factor
    :proper_time       # Accumulated proper time
  ]

  @c 1.0  # Speed of light in natural units

  def new(initial_position \\ {0.0, 0.0, 0.0}) do
    %__MODULE__{
      worldline: [{0.0, initial_position}],
      velocity: {0.0, 0.0, 0.0},
      gamma: 1.0,
      proper_time: 0.0
    }
  end

  def accelerate(st, acceleration, dt) do
    # Update velocity with relativistic addition
    {vx, vy, vz} = st.velocity
    {ax, ay, az} = acceleration
    
    vx = relativistic_velocity_add(vx, ax * dt)
    vy = relativistic_velocity_add(vy, ay * dt)
    vz = relativistic_velocity_add(vz, az * dt)
    
    v_squared = vx*vx + vy*vy + vz*vz
    gamma = if v_squared < @c*@c do
      1.0 / :math.sqrt(1.0 - v_squared / (@c*@c))
    else
      :infinity
    end
    
    # Update proper time (dτ = dt / γ)
    proper_time = st.proper_time + dt / gamma
    
    # Update worldline
    {t, {x, y, z}} = List.last(st.worldline)
    new_event = {t + dt, {x + vx * dt, y + vy * dt, z + vz * dt}}
    worldline = st.worldline ++ [new_event]
    
    %{st | velocity: {vx, vy, vz}, gamma: gamma, proper_time: proper_time, worldline: worldline}
  end

  def spacetime_interval(event1, event2) do
    {t1, {x1, y1, z1}} = event1
    {t2, {x2, y2, z2}} = event2
    
    dt = t2 - t1
    dx = x2 - x1
    dy = y2 - y1
    dz = z2 - z1
    
    # Minkowski metric: ds² = -c²dt² + dx² + dy² + dz²
    -@c*@c * dt*dt + dx*dx + dy*dy + dz*dz
  end

  def is_timelike?(event1, event2) do
    spacetime_interval(event1, event2) < 0
  end

  def is_lightlike?(event1, event2) do
    spacetime_interval(event1, event2) == 0
  end

  def is_spacelike?(event1, event2) do
    spacetime_interval(event1, event2) > 0
  end

  def proper_time_between(event1, event2) do
    if is_timelike?(event1, event2) do
      :math.sqrt(-spacetime_interval(event1, event2)) / @c
    else
      0.0
    end
  end

  defp relativistic_velocity_add(u, v) do
    (u + v) / (1.0 + u * v / (@c * @c))
  end
end

defmodule NeuralMeshLite.Thermodynamics do
  @moduledoc """
  Thermodynamic computing: minimize entropy production.
  Agents evolve toward reversible computation for energy efficiency.
  """
  defstruct [
    :entropy_produced,
    :energy_consumed,
    :reversible_operations,
    :irreversible_operations,
    :temperature,      # Computational temperature (noise level)
    :efficiency        # Reversible fraction
  ]

  def new do
    %__MODULE__{
      entropy_produced: 0.0,
      energy_consumed: 0.0,
      reversible_operations: 0,
      irreversible_operations: 0,
      temperature: 0.1,
      efficiency: 0.5
    }
  end

  def compute(td, operation, input_size) do
    # Landauer's principle: erasing 1 bit costs kT ln 2 energy
    k = 1.0  # Boltzmann constant (normalized)
    bit_energy = k * td.temperature * :math.log(2)
    
    case operation do
      :reversible ->
        # Reversible operations (e.g., Toffoli gate) dissipate no energy
        energy = 0.0
        entropy = 0.0
        rev = 1
        irrev = 0
        
      :irreversible ->
        # Irreversible operations (e.g., AND gate) erase bits
        bits_erased = input_size
        energy = bits_erased * bit_energy
        entropy = bits_erased * k * :math.log(2)
        rev = 0
        irrev = 1
        
      :mixed ->
        # Partially reversible
        reversible_fraction = td.efficiency
        bits_erased = input_size * (1.0 - reversible_fraction)
        energy = bits_erased * bit_energy
        entropy = bits_erased * k * :math.log(2)
        rev = 0
        irrev = 1
    end
    
    %{td | 
      entropy_produced: td.entropy_produced + entropy,
      energy_consumed: td.energy_consumed + energy,
      reversible_operations: td.reversible_operations + rev,
      irreversible_operations: td.irreversible_operations + irrev
    }
  end

  def evolve_efficiency(td, fitness_gradient) do
    # Positive gradient -> increase efficiency
    new_efficiency = max(0.1, min(0.99, td.efficiency + fitness_gradient * 0.01))
    # Temperature decreases with efficiency (cooler computation)
    new_temp = max(0.01, td.temperature * (1.0 - fitness_gradient * 0.01))
    
    %{td | efficiency: new_efficiency, temperature: new_temp}
  end

  def free_energy(td, internal_energy, entropy) do
    # F = U - TS
    internal_energy - td.temperature * entropy
  end
end

defmodule NeuralMeshLite.Topology do
  @moduledoc """
  Topological data analysis using persistent homology.
  Understands shape of high-dimensional data.
  """
  defstruct [
    :simplicial_complex,
    :persistence_diagram,
    :betti_numbers
  ]

  def new do
    %__MODULE__{
      simplicial_complex: [],
      persistence_diagram: %{},
      betti_numbers: %{0 => 0, 1 => 0, 2 => 0}
    }
  end

  def build_vietoris_rips(points, max_dim \\ 2, max_radius \\ 1.0) do
    # Build Vietoris-Rips complex from point cloud
    n = length(points)
    
    # 0-simplices (vertices)
    vertices = Enum.map(0..(n-1), &[&1])
    
    # 1-simplices (edges) - pairs within radius
    edges = for i <- 0..(n-1), j <- (i+1)..(n-1) do
      dist = euclidean(Enum.at(points, i), Enum.at(points, j))
      if dist <= max_radius, do: [i, j, dist]
    end |> Enum.reject(&is_nil/1)
    
    # 2-simplices (triangles) - triples with all edges present
    triangles = if max_dim >= 2 do
      for i <- 0..(n-1), j <- (i+1)..(n-1), k <- (j+1)..(n-1) do
        edge_ij = Enum.find(edges, fn e -> (Enum.at(e,0)==i and Enum.at(e,1)==j) or (Enum.at(e,0)==j and Enum.at(e,1)==i) end)
        edge_jk = Enum.find(edges, fn e -> (Enum.at(e,0)==j and Enum.at(e,1)==k) or (Enum.at(e,0)==k and Enum.at(e,1)==j) end)
        edge_ki = Enum.find(edges, fn e -> (Enum.at(e,0)==k and Enum.at(e,1)==i) or (Enum.at(e,0)==i and Enum.at(e,1)==k) end)
        if edge_ij && edge_jk && edge_ki do
          max_filtration = max(max(Enum.at(edge_ij,2), Enum.at(edge_jk,2)), Enum.at(edge_ki,2))
          [i, j, k, max_filtration]
        end
      end |> Enum.reject(&is_nil/1)
    else
      []
    end
    
    complex = %{
      vertices: vertices,
      edges: edges,
      triangles: triangles
    }
    
    %__MODULE__{simplicial_complex: complex}
  end

  def compute_persistence(topo) do
    # Compute persistence diagram (simplified)
    edges = topo.simplicial_complex.edges |> Enum.sort_by(&Enum.at(&1,2))
    
    # Birth-death pairs for H0 (connected components)
    h0_pairs = compute_h0_persistence(edges, length(topo.simplicial_complex.vertices))
    
    # Birth-death pairs for H1 (loops)
    h1_pairs = compute_h1_persistence(topo.simplicial_complex.triangles, edges)
    
    persistence = %{0 => h0_pairs, 1 => h1_pairs}
    
    betti = %{
      0 => length(Enum.filter(h0_pairs, fn {_, death} -> death == :infinity end)),
      1 => length(Enum.filter(h1_pairs, fn {_, death} -> death == :infinity end))
    }
    
    %{topo | persistence_diagram: persistence, betti_numbers: betti}
  end

  def topological_features(topo) do
    %{
      connected_components: topo.betti_numbers[0],
      loops: topo.betti_numbers[1],
      total_persistence: total_persistence(topo.persistence_diagram)
    }
  end

  defp euclidean(p1, p2) do
    :math.sqrt(Enum.zip(p1, p2) |> Enum.map(fn {a,b} -> (a-b)*(a-b) end) |> Enum.sum())
  end

  defp compute_h0_persistence(edges, n) do
    # Union-find for connected components
    parent = Enum.to_list(0..(n-1))
    
    Enum.reduce(edges, [], fn edge, acc ->
      [u, v, weight] = edge
      root_u = find(parent, u)
      root_v = find(parent, v)
      if root_u != root_v do
        parent = union(parent, root_u, root_v)
        [{0.0, weight} | acc]
      else
        acc
      end
    end) ++ (for _ <- 1..n, do: {0.0, :infinity})
  end

  defp compute_h1_persistence(triangles, _edges) do
    # Simplified: count triangles as H1 generators when they form boundaries
    Enum.map(triangles, fn tri -> {Enum.at(tri, 3), :infinity} end)
  end

  defp find(parent, x) do
    if Enum.at(parent, x) == x, do: x, else: find(parent, Enum.at(parent, x))
  end

  defp union(parent, x, y) do
    List.replace_at(parent, x, y)
  end

  defp total_persistence(pd) do
    (pd[0] ++ pd[1])
    |> Enum.map(fn {birth, death} -> if death == :infinity, do: 1.0, else: death - birth end)
    |> Enum.sum()
  end
end

      reward_sensitivity: state.dopamine

      serotonin: state.serotonin * (1 - state.decay_rate),
      acetylcholine: state.acetylcholine * (1 - state.decay_rate),
      noradrenaline: state.noradrenaline * (1 - state.decay_rate)
    }
    schedule_decay()
    {:noreply, state}
  end

defmodule NeuralMeshLite.CausalInference do
  @moduledoc """
  Causal graph inference from observational and interventional data.
  Uses do-calculus for reasoning about interventions.
  """
  defstruct [
    :graph,            # %{node => [children]}
    :parents,          # %{node => [parents]}
    :probabilities,    # %{node => %{parent_values => distribution}}
    :interventions     # History of do-operations
  ]

  def new do
    %__MODULE__{
      graph: %{},
      parents: %{},
      probabilities: %{},
      interventions: []
    }
  end

  def observe(ci, node, value, parent_values \\ %{}) do
    # Update conditional probability P(node | parents)
    key = parent_values
    probs = Map.get(ci.probabilities, node, %{})
    node_probs = Map.update(probs, key, %{value => 1}, fn dist ->
      Map.update(dist, value, 1, &(&1 + 1))
    end)
    
    %{ci | probabilities: Map.put(ci.probabilities, node, node_probs)}
  end

  def intervene(ci, node, value) do
    # do(node = value) - surgically set node, breaking incoming edges
    intervention = %{node: node, value: value, timestamp: DateTime.utc_now()}
    %{ci | interventions: [intervention | ci.interventions]}
  end

  def learn_structure(ci, data) do
    # PC algorithm (simplified) for causal discovery
    nodes = Map.keys(data) |> Enum.uniq()
    
    # Initialize fully connected undirected graph
    graph = Map.new(nodes, fn n -> {n, nodes -- [n]} end)
    
    # Remove edges based on conditional independence tests
    graph = Enum.reduce(nodes, graph, fn x, g ->
      Enum.reduce(Map.get(g, x), g, fn y, g2 ->
        if independent?(x, y, [], data) do
          # Remove edge x-y
          Map.update!(g2, x, &List.delete(&1, y))
          |> Map.update!(y, &List.delete(&1, x))
        else
          g2
        end
      end)
    end)
    
    # Orient edges using v-structures
    parents = orient_edges(graph, data)
    
    %{ci | graph: graph, parents: parents}

    infer_probability(truncated_graph, ci.probabilities, target, evidence)
  end

defmodule NeuralMeshLite.ActiveInference do
  @moduledoc """
  Active inference: agents minimize variational free energy.
  Balances exploration (epistemic value) and exploitation (pragmatic value).
  """
  defstruct [
    :generative_model,   # P(o,s) - joint probability of observations and states
    :variational_belief, # Q(s) - approximate posterior over states
    :free_energy,
    :expected_free_energy
  ]

  def new(num_states \\ 10, num_observations \\ 10) do
    # Initialize generative model with random transitions
    transitions = for _ <- 1..num_states, do: random_prob_vector(num_states)
    likelihood = for _ <- 1..num_states, do: random_prob_vector(num_observations)
    prior = random_prob_vector(num_states)
    
    %__MODULE__{
      generative_model: %{A: likelihood, B: transitions, D: prior},
      variational_belief: prior,
      free_energy: 0.0,
      expected_free_energy: %{}
    }
  end

  def infer_state(ai, observation) do
    # Variational inference: update Q(s) to minimize free energy
    # F = D_KL[Q(s) || P(s)] - E_Q[log P(o|s)]
    
    model = ai.generative_model
    
    # Compute likelihood-weighted belief
    likelihoods = for s <- 0..(length(model.A)-1) do
      Enum.at(Enum.at(model.A, s), observation)
    end
    
    # Update belief (simplified: Bayes rule)
    new_belief = Enum.zip(ai.variational_belief, likelihoods)
    |> Enum.map(fn {prior, lik} -> prior * lik end)
    |> normalize()
    
    # Compute free energy
    fe = free_energy(new_belief, ai.variational_belief, likelihoods)
    
    %{ai | variational_belief: new_belief, free_energy: fe}
  end

  def select_action(ai, available_actions) do
    # Compute expected free energy for each action
    # G(π) = -E_Q(o,s|π)[log P(o|s) + log Q(s|o,π) - log Q(s|π)]
    #       = epistemic value + pragmatic value
    
    action_g_values = Enum.map(available_actions, fn action ->
      expected_states = predict_next_states(ai, action)
      
      # Epistemic value: expected information gain
      epistemic = compute_epistemic_value(ai, expected_states)
      
      # Pragmatic value: expected reward
      pragmatic = compute_pragmatic_value(expected_states)
      
      {action, epistemic + pragmatic}
    end)
    
    # Softmax selection
    {action, _} = softmax_select(action_g_values)
    action
  end

  def update_model(ai, state, action, observation, reward) do
    # Update generative model based on experience
    model = ai.generative_model
    
    # Update transition probabilities (B)
    new_B = update_transitions(model.B, state, action, reward)
    
    # Update likelihood (A)
    new_A = update_likelihood(model.A, state, observation)
    
    %{ai | generative_model: %{model | A: new_A, B: new_B}}
  end

  defp random_prob_vector(n) do
    vec = for _ <- 1..n, do: :rand.uniform()
    normalize(vec)
  end

  defp normalize(vec) do
    total = Enum.sum(vec)
    if total > 0, do: Enum.map(vec, &(&1 / total)), else: vec
  end

  defp free_energy(q, prior, likelihood) do
    # F = Σ q(s) log(q(s)/prior(s)) - Σ q(s) log likelihood
    kl = Enum.zip(q, prior) |> Enum.map(fn {q_i, p_i} -> if q_i > 0 and p_i > 0, do: q_i * :math.log(q_i/p_i), else: 0 end) |> Enum.sum()
    expected_ll = Enum.zip(q, likelihood) |> Enum.map(fn {q_i, l_i} -> q_i * :math.log(max(l_i, 1.0e-10)) end) |> Enum.sum()
    kl - expected_ll
  end

  defp predict_next_states(_ai, _action), do: random_prob_vector(10)
  defp compute_epistemic_value(_ai, _states), do: :rand.uniform()
  defp compute_pragmatic_value(_states), do: :rand.uniform()
  defp softmax_select(list) do
    exp_sum = Enum.map(list, fn {_, v} -> :math.exp(v) end)
    total = Enum.sum(exp_sum)
    r = :rand.uniform() * total
    Enum.reduce_while(Enum.zip(list, exp_sum), {nil, r}, fn {{a, _}, e}, {_, rem} ->
      if rem <= e, do: {:halt, {a, rem}}, else: {:cont, {nil, rem - e}}
    end)
  end

    # Simplified update
    b
  end
  defp update_likelihood(a, state, obs) do

    a
  end
end

    red_team = [%{id: mind_id, generator: generator_fn, fitness: 0.0, wins: 0, losses: 0} | state.red_team]
    {:noreply, %{state | red_team: red_team}}
  end

  def handle_cast({:register_blue, mind_id, critic_fn}, state) do
    blue_team = [%{id: mind_id, critic: critic_fn, fitness: 0.0, detections: 0, misses: 0} | state.blue_team]
    {:noreply, %{state | blue_team: blue_team}}
  end

  def handle_cast(:evolve, state) do
    if length(state.red_team) >= 2 and length(state.blue_team) >= 2 do
      # Tournament between red and blue
      {red_fitnesses, blue_fitnesses} = run_tournament(state.red_team, state.blue_team)
      
      red_team = update_fitnesses(state.red_team, red_fitnesses)
      blue_team = update_fitnesses(state.blue_team, blue_fitnesses)
      
      # Evolve both teams
      red_team = evolve_population(red_team, :generator)
      blue_team = evolve_population(blue_team, :critic)
      
      Logger.info("Adversarial evolution: Gen #{state.generation} - Red avg fitness: #{avg_fitness(red_team)}, Blue avg: #{avg_fitness(blue_team)}")
      
      {:noreply, %{state | red_team: red_team, blue_team: blue_team, generation: state.generation + 1}}
    else
      {:noreply, state}
    end
  end

  @impl true
  def handle_call({:evaluate_red, mind_id, sample}, _from, state) do
    red = Enum.find(state.red_team, &(&1.id == mind_id))
    if red do
      # Find best blue critic to evaluate against
      best_blue = Enum.max_by(state.blue_team, & &1.fitness, fn -> nil end)
      if best_blue do
        generated = red.generator.(sample)
        detected = best_blue.critic.(generated)
        {:reply, %{generated: generated, detected: detected}, state}
      else
        {:reply, %{generated: red.generator.(sample), detected: false}, state}
      end
    else
      {:reply, nil, state}
    end
  end

  defp run_tournament(red_team, blue_team) do
    # Each red generates samples; each blue tries to detect
    red_scores = Enum.map(red_team, fn red ->
      Enum.map(blue_team, fn blue ->
        # Red wins if blue fails to detect
        sample = red.generator.(:test_input)
        detected = blue.critic.(sample)
        if detected, do: 0, else: 1
      end)
    end)
    
    red_fitnesses = Enum.map(red_scores, fn scores -> Enum.sum(scores) / length(scores) end)
    
    blue_scores = Enum.map(blue_team, fn blue ->
      Enum.map(red_team, fn red ->
        sample = red.generator.(:test_input)
        detected = blue.critic.(sample)
        if detected, do: 1, else: 0
      end)
    end)
    
    blue_fitnesses = Enum.map(blue_scores, fn scores -> Enum.sum(scores) / length(scores) end)
    
    {red_fitnesses, blue_fitnesses}
  end

  defp update_fitnesses(team, fitnesses) do
    Enum.zip(team, fitnesses)
    |> Enum.map(fn {member, fitness} -> %{member | fitness: fitness} end)
  end

  defp evolve_population(team, _type) do
    # Keep top performers, mutate others
    sorted = Enum.sort_by(team, & &1.fitness, :desc)
    elite_count = max(1, div(length(team), 5))
    elites = Enum.take(sorted, elite_count)
    
    offspring = for _ <- 1..(length(team) - elite_count) do
      parent = Enum.random(elites)
      # Mutate the generator/critic function (simplified: return new version)
      %{parent | id: UUID.uuid4(), fitness: 0.0}
    end
    
    elites ++ offspring
  end

  defp avg_fitness(team) do
    if team == [], do: 0.0, else: Enum.map(team, & &1.fitness) |> Enum.sum() / length(team)
  end

    x = max(0, min(sx-1, round(x)))
    y = max(0, min(sy-1, round(y)))
    z = max(0, min(sz-1, round(z)))

      List.update_at(yz, y, fn zs ->
        List.update_at(zs, z, &(&1 + amount))
      end)
    end)
    
    sources = Map.put(state.sources, mind_id, {{x, y, z}, type, amount})

  end

    {sx, sy, sz} = state.dimensions
    x = max(0, min(sx-1, round(x)))
    y = max(0, min(sy-1, round(y)))
    z = max(0, min(sz-1, round(z)))

    {:reply, concentration, state}
  end

  def handle_call({:gradient, {x, y, z}}, _from, state) do
    {sx, sy, sz} = state.dimensions
    x = max(1, min(sx-2, round(x)))
    y = max(1, min(sy-2, round(y)))
    z = max(1, min(sz-2, round(z)))
    
    c = fn dx, dy, dz -> 
      state.field |> Enum.at(x+dx) |> Enum.at(y+dy) |> Enum.at(z+dz)
    end
    
    # Central difference gradient
    gx = (c.(1,0,0) - c.(-1,0,0)) / 2.0
    gy = (c.(0,1,0) - c.(0,-1,0)) / 2.0
    gz = (c.(0,0,1) - c.(0,0,-1)) / 2.0
    
    {:reply, {gx, gy, gz}, state}
  end

  @impl true
  def handle_info(:diffuse, state) do
    # Simple diffusion: average with neighbors
    {sx, sy, sz} = state.dimensions
    new_field = for x <- 0..(sx-1) do
      for y <- 0..(sy-1) do
        for z <- 0..(sz-1) do
          current = state.field |> Enum.at(x) |> Enum.at(y) |> Enum.at(z)
          neighbors = for dx <- -1..1, dy <- -1..1, dz <- -1..1,
                          x+dx >= 0, x+dx < sx,
                          y+dy >= 0, y+dy < sy,
                          z+dz >= 0, z+dz < sz,
                          not (dx==0 and dy==0 and dz==0) do
            state.field |> Enum.at(x+dx) |> Enum.at(y+dy) |> Enum.at(z+dz)
          end
          avg_neighbor = if neighbors == [], do: current, else: Enum.sum(neighbors) / length(neighbors)
          current * (1 - state.diffusion_rate) + avg_neighbor * state.diffusion_rate
        end
      end
    end
    
    # Apply decay
    new_field = new_field |> Enum.map(fn yz -> yz |> Enum.map(fn zs -> zs |> Enum.map(&(&1 * (1 - state.decay_rate))) end) end)
    
    schedule_diffusion()
    {:noreply, %{state | field: new_field}}
  end

  defp schedule_diffusion, do: Process.send_after(self(), :diffuse, 1_000)
end

    end

defmodule NeuralMeshLite.Metacognition do
  @moduledoc """
  Meta-cognitive reflection: agents think about their own thinking.
  Tracks confidence, uncertainty, and strategy effectiveness.
  """
  defstruct [
    :confidence_history,
    :uncertainty_estimates,
    :strategy_performance,
    :self_model,
    :reflection_log
  ]

  def new do
    %__MODULE__{
      confidence_history: [],
      uncertainty_estimates: %{},
      strategy_performance: %{},
      self_model: %{},
      reflection_log: []
    }
  end

  def reflect(meta, decision, outcome, confidence, context) do
    # Update confidence calibration
    calibrated = abs(confidence - outcome) |> then(&(1.0 - &1))
    confidence_history = [{confidence, outcome, calibrated} | meta.confidence_history] |> Enum.take(100)
    
    # Update strategy performance
    strategy = Map.get(context, :strategy, :unknown)
    current = Map.get(meta.strategy_performance, strategy, {0.0, 0})
    {sum, n} = current
    new_perf = Map.put(meta.strategy_performance, strategy, {sum + outcome, n + 1})
    
    # Update self-model (predicts own performance)
    self_model = update_self_model(meta.self_model, context, outcome)
    
    # Log reflection
    reflection = %{
      timestamp: DateTime.utc_now(),
      decision: decision,
      outcome: outcome,
      confidence: confidence,
      calibration_error: 1.0 - calibrated,
      insight: generate_insight(meta, decision, outcome, confidence)
    }
    reflection_log = [reflection | meta.reflection_log] |> Enum.take(50)
    
    %{meta | 
      confidence_history: confidence_history,
      strategy_performance: new_perf,
      self_model: self_model,
      reflection_log: reflection_log
    }
  end

  def estimate_uncertainty(meta, context) do
    # Predict uncertainty for given context
    if meta.self_model == %{} do
      0.5
    else
      predict_uncertainty(meta.self_model, context)
    end
  end

  def best_strategy(meta) do
    if meta.strategy_performance == %{} do
      nil
    else
      {best, _} = Enum.max_by(meta.strategy_performance, fn {_, {sum, n}} -> sum / n end)
      best
    end
  end

  def calibration_curve(meta) do
    # Group predictions by confidence bin
    bins = for i <- 0..9, do: i/10
    Enum.map(bins, fn bin ->
      in_bin = Enum.filter(meta.confidence_history, fn {conf, _, _} -> 
        conf >= bin and conf < bin + 0.1
      end)
      if in_bin == [] do
        {bin, 0.0, 0}
      else
        avg_outcome = Enum.map(in_bin, fn {_, out, _} -> out end) |> Enum.sum() / length(in_bin)
        {bin, avg_outcome, length(in_bin)}
      end
    end)
  end

  defp update_self_model(model, context, outcome) do
    # Simple linear model update
    features = context_to_features(context)
    if model == %{} do
      %{weights: features, bias: outcome}
    else
      lr = 0.01
      prediction = predict(model, features)
      error = outcome - prediction
      new_weights = Enum.zip(model.weights, features)
      |> Enum.map(fn {w, f} -> w + lr * error * f end)
      new_bias = model.bias + lr * error
      %{weights: new_weights, bias: new_bias}
    end
  end

  defp predict_uncertainty(model, context) do
    features = context_to_features(context)
    pred = predict(model, features)
    # Uncertainty = 1 - |prediction - 0.5| * 2 (max at 0.5)
    1.0 - abs(pred - 0.5) * 2.0
  end

  defp predict(model, features) do
    if model == %{} do
      0.5
    else
      dot = Enum.zip(model.weights, features) |> Enum.map(fn {w, f} -> w * f end) |> Enum.sum()
      sigmoid(dot + model.bias)
    end
  end

  defp context_to_features(context) do
    [
      Map.get(context, :complexity, 0.5),
      Map.get(context, :novelty, 0.5),
      Map.get(context, :energy, 0.5)
    ]
  end

  defp sigmoid(x), do: 1.0 / (1.0 + :math.exp(-x))

  defp generate_insight(meta, decision, outcome, confidence) do
    overconfident = confidence > 0.7 and outcome < 0.3
    underconfident = confidence < 0.3 and outcome > 0.7
    
    cond do
      overconfident -> "I was overconfident in #{decision}. Need to calibrate down."
      underconfident -> "I was underconfident in #{decision}. Trust my instincts more."
      true -> "Calibration was good for #{decision}."
    end
  end
end
